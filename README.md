def is_anagram(word1, word2):
    word1 = word1.lower().replace(" ", "")
    word2 = word2.lower().replace(" ", "")

    dict_1 = {}
    dict_2 = {}

    for char in word1:
        dict_1[char] = dict_1.get(char, 0) + 1

    for char in word2:
        dict_2[char] = dict_2.get(char, 0) + 1

    return dict_1 == dict_2
